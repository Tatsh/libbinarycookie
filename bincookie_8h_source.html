145static inline bincookie_t *const bincookie_init_file(FILE *fin) {
146 const long last_pos = ftell(fin);
147 rewind(fin);
148 char magic[4];
149 size_t read = fread(magic, sizeof(magic), 1, fin);
150 bool is_cook = magic[0] == 'c' && magic[1] == 'o' && magic[2] == 'o' && magic[3] == 'k';
151 if (read != 1 || !is_cook) {
152 errno = EIO;
153 return NULL;
154 }
155 // Read entire file
156 fseek(fin, 0, SEEK_END);
157 size_t num_bytes = (size_t)ftell(fin);
158 bincookie_t *cook = (bincookie_t *)malloc(num_bytes);
159 // LCOV_EXCL_START
160 if (cook == NULL) {
161 return NULL;
162 } // LCOV_EXCL_STOP
163 memset(cook, 0, num_bytes);
164 rewind(fin);
165 if (fread(cook, num_bytes, 1, fin) != 1) {
166 return NULL;
167 }
168 cook->num_pages = __builtin_bswap32(cook->num_pages);
169 // Fix page size numbers
170 for (uint32_t i = 0; i < cook->num_pages; i++) {
171 *(cook->page_sizes + i) = __builtin_bswap32(*(cook->page_sizes + i));
172 }
173 fseek(fin, last_pos, SEEK_SET);
174 return cook;
175}

196static inline bincookie_page_t *const bincookie_iter_pages(const bincookie_t *bc,
197 bincookie_iter_state_t *const state) {
198 if (state->page_offset == 0) {
199 state->page_offset = (uint32_t)(4 + sizeof(uint32_t) + (sizeof(uint32_t) * bc->num_pages));
200 }
201 if (state->page_index < bc->num_pages) {
202 bincookie_page_t *page = (bincookie_page_t *)((char *)bc + state->page_offset);
203 state->page_offset += bc->page_sizes[state->page_index];
204 state->page_index++;
